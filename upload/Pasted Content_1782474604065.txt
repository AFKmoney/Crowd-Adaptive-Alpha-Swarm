from dataclasses import dataclass, field
from typing import Dict, Optional
from enum import Enum
import time

class Side(Enum):
    BUY = "BUY"
    SELL = "SELL"

@dataclass
class MarketEvent:
    symbol: str
    timestamp: float = field(default_factory=time.time)
    price: float = 0.0
    funding_rate: float = 0.0
    ls_ratio: float = 0.0  
    open_interest: float = 0.0
    volatility_atr_bps: float = 0.0
    spoofing_imbalance: float = 0.0  # NOUVEAU: -1.0 (Bids annulés -> Bearish), +1.0 (Asks annulés -> Bullish)

@dataclass(frozen=True)
class CrowdPositioningEvent:
    symbol: str
    timestamp: float
    crowd_score: float         # [-1, +1] : neg = crowd is short, pos = crowd is long
    conviction: float          # 0..1
    horizon: str               # "minutes" | "hours" | "days"
    components: Dict[str, float]
    regime_hint: str           # "cascade_imminent" | "euphoria" | "fear" | "neutral"
    expected_move_bps: float

@dataclass
class SignalEvent:
    symbol: str
    side: Side
    confidence: float          # 0..1
    expected_holding_period_bars: int
    take_profit_bps: float
    stop_loss_bps: float
    rationale: str
    timestamp: float = field(default_factory=time.time)

    @dataclass
    class OrderEvent:
        symbol: str
        side: Side
        size_usd: float
        order_type: str = "MARKET"
        price: Optional[float] = None
        is_grid: bool = False       # NOUVEAU: Exécution Maker-Grid
        grid_levels: int = 1

        order = OrderEvent(
            symbol=signal.symbol,
            side=signal.side,
            size_usd=position_size_usd,
            order_type="LIMIT" if is_grid else "MARKET",
            price=market.price,
            is_grid=is_grid,
            grid_levels=grid_levels
        )
        logger.info(f"Signal Approved: Sized at ${position_size_usd:.2f} USD (Grid: {is_grid})")

import asyncio
from typing import Dict, Any

from skytrader.config import settings
from skytrader.core.logger import get_logger
from skytrader.core.events import MarketEvent

logger = get_logger("OKXFeed")

class OKXClient:
    def __init__(self):
        self.exchange = ccxt.okx({
            'apiKey': settings.OKX_API_KEY,
            'secret': settings.OKX_SECRET_KEY,
            'password': settings.OKX_PASSPHRASE,
            'enableRateLimit': True,
        })
        if settings.USE_TESTNET:
            logger.info("Connecting to OKX TESTNET")
            self.exchange.set_sandbox_mode(True)
        else:
            logger.warning("Connecting to OKX MAINNET - REAL FUNDS AT RISK")
            
        self._ob_state = {}  # Pour traquer le Spoofing

    async def fetch_market_state(self, symbol: str) -> MarketEvent:
        """Fetch unified market state (Price, Funding, L/S Ratio)"""
        try:
            # 1. Ticker (Price)
            ticker = await self.exchange.fetch_ticker(symbol)
            price = ticker['last']
            
            # 2. Funding Rate
            # OKX Swap funding rate
            funding_info = await self.exchange.fetch_funding_rate(symbol)
            funding_rate = funding_info.get('fundingRate', 0.0)
            
            # 3. L/S Ratio (Approximation)
            currency = symbol.split('-')[0]
            try:
                ls_data = await self.exchange.public_get_rubik_stat_contracts_long_short_account_ratio({
                    'ccy': currency,
                    'period': '5m'
                })
                ls_ratio = float(ls_data['data'][0][1])
            except Exception as e:
                ls_ratio = 1.0
                
            # 4. ATR / Volatility (Nouvelle Arme de Précision)
            try:
                ohlcv = await self.exchange.fetch_ohlcv(symbol, timeframe='5m', limit=14)
                atr = sum([candle[2] - candle[3] for candle in ohlcv]) / len(ohlcv) if ohlcv else 0.0
                volatility_atr_bps = (atr / price) * 10000 if price else 0.0
            except Exception:
                volatility_atr_bps = 50.0 # Default fallback

            # --- 5. NOUVEAU: WHALE WAKE DETECTOR (Spoofing) ---
            spoofing_imbalance = 0.0
            try:
                ob = await self.exchange.fetch_order_book(symbol, limit=20)
                bids_vol = sum(b[1] for b in ob['bids'])
                asks_vol = sum(a[1] for a in ob['asks'])
                
                if symbol in self._ob_state:
                    prev_bids, prev_asks = self._ob_state[symbol]
                    # Si le volume a baissé de plus de 40% soudainement
                    bid_drop = (prev_bids - bids_vol) / prev_bids if prev_bids > 0 else 0
                    ask_drop = (prev_asks - asks_vol) / prev_asks if prev_asks > 0 else 0
                    
                    if bid_drop > 0.40:
                        spoofing_imbalance = -1.0  # Baleine a retiré ses achats -> Bearish
                        logger.warning(f"🐋 WHALE WAKE: Faux mur d'achat annulé sur {symbol} ! Bids -{bid_drop*100:.1f}%")
                    elif ask_drop > 0.40:
                        spoofing_imbalance = 1.0   # Baleine a retiré ses ventes -> Bullish
                        logger.warning(f"🐋 WHALE WAKE: Faux mur de vente annulé sur {symbol} ! Asks -{ask_drop*100:.1f}%")
                        
                self._ob_state[symbol] = (bids_vol, asks_vol)
            except Exception as e:
                pass
                
            return MarketEvent(
                symbol=symbol,
                price=price,
                funding_rate=funding_rate,
                ls_ratio=ls_ratio,
                volatility_atr_bps=volatility_atr_bps,
                spoofing_imbalance=spoofing_imbalance
            )
        except Exception as e:
            logger.error(f"Error fetching market state for {symbol}: {str(e)}")
            return MarketEvent(symbol=symbol)

    async def close(self):
        await self.exchange.close()
