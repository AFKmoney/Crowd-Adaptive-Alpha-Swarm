class CrowdPositioningEvent:
    symbol: str
    timestamp: str
    # Score composite ∈ [-1, +1] : négatif = foule overcrowded SHORT,

class ContrarianAgent(AlphaAgent):
    """Fade les extrêmes de positioning. Le cœur de la thèse."""
    
    def on_positioning(self, event: CrowdPositioningEvent) -> List[SignalEvent]:
        if abs(event.crowd_score) < EXTREME_THRESHOLD:  # 0.5 par défaut
            return []  # pas un extrême → on ne fade pas
        # FADER la foule
        side = Side.SELL if event.crowd_score > 0 else Side.BUY
        confidence = event.conviction * 0.85  # cap à 0.85 (jamais certain)
        return [SignalEvent(

            confidence=confidence,
            expected_holding_period_bars=HORIZON_BARS[event.horizon],
            # TP/SL asymétriques : on vise la cascade (grand TP), stop tight

            take_profit_bps=event.expected_move_bps,
            stop_loss_bps=event.expected_move_bps * 0.3,
            rationale=f"Fade crowd {event.regime_hint}: score={event.crowd_score:+.2f}",
